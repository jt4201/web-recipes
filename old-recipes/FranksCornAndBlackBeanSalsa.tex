\renewcommand{\mytitle}{Corn and Black Bean Salsa (Franks)}
\renewcommand{\myurl}{https://www.franksredhot.com/en-us/recipes/corn-black-bean-salsa}

\sbox{\ingredients}{%
  \begin{minipage}[t]{0.32\textwidth}
    {\sffamily\bfseries Ingredients}\vspace{0.5ex}
    \begin{itemize}
      \item \nicefrac{1}{3} cup olive oil
      \item 3 tablespoons Frank's RedHot® Original Cayenne Pepper Hot Sauce
      \item 3 tablespoons red wine vinegar
      \item 2 tablespoons Gourmet Garden™ Lightly Dried Cilantro
      \item 1 teaspoon Gourmet Garden™ Garlic Stir-In Paste
      \item \nicefrac{1}{2} teaspoon McCormick® Chili Powder
      \item \nicefrac{1}{4} teaspoon salt
      \item \nicefrac{1}{4} teaspoon McCormick® Pure Ground Black Pepper
      \item 1 can (15 \nicefrac{1}{2} ounces) black beans, drained and rinsed
      \item 1 can (14 \nicefrac{1}{4} ounces) whole kernel corn, drained
      \item 1 medium plum tomato, chopped
      \item 2 green onions, thinly sliced
    \end{itemize}
  \end{minipage}
}

\sbox{\method}{%
  \begin{minipage}[t]{0.55\textwidth}
    {\sffamily\bfseries Method}\vspace{0.5ex}
    \begin{enumerate}
      \item MIX oil, RedHot Sauce, vinegar, cilantro, garlic, chili powder, salt and pepper in a large bowl with wire whisk until well-blended.
      \item ADD beans, corn, tomato and onions, tossing to coat with dressing. Cover and refrigerate at least 30 minutes or until ready to serve. Garnish with fresh lime wedges and additional Lightly Dried Cilantro, if desired. Serve with tortilla chips or as a topping for tacos, burritos and nachos.
    \end{enumerate}
  \end{minipage}
}
