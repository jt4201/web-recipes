\sbox{\Ingredients}{%
  \begin{minipage}[t]{\the\dimexpr 0.4\textwidth - 2em}
    \par\mySubtitle{Ingredients}
    \par\begin{description}
      \item[Salad] \hfill
        \begin{itemize}
          \item 1 cup uncooked quinoa
          \item 2 cups water or broth
          \item 2 roma tomatoes diced
          \item 1 ripe avocado diced
          \item 15 ounces black beans 1 can, rinsed and drained
          \item 12 ounces corn 1 can drained
          \item 1 bell pepper diced
          \item \nicefrac{1}{4} cup green onion diced
          \item 1 jalapeno pepper seeded and diced
          \item \nicefrac{1}{3} cup cilantro chopped
        \end{itemize}
      \item[Dressing] \hfill
        \begin{itemize}
          \item \nicefrac{1}{4} cup vegetable oil
          \item 1 \nicefrac{1}{2} tablespoon lime juice
          \item 1 teaspoon sugar
          \item 1 teaspoon cumin
          \item \nicefrac{1}{4} teaspoon garlic powder
          \item \nicefrac{1}{2} tablespoon red wine vinegar
          \item \nicefrac{1}{8} teaspoon salt
          \item \nicefrac{1}{4} teaspoon pepper
        \end{itemize}
    \end{description}
  \end{minipage}
}

\sbox{\Instructions}{%
  \begin{minipage}[t]{\the\dimexpr 0.6\textwidth - 2em}
    \par\mySubtitle{Instructions}
    \par\begin{enumerate}
      \item Cook quinoa according to package directions and allow to cool.
      \item Combine quinoa, tomatoes, avocado, black beans, corn, bell pepper, green
        onion, jalapeno and cilantro in a large bowl.
      \item In a small bowl, combine all dressing ingredients and mix well. Pour over
        vegetables and toss to coat. Add quinoa and combine.
      \item Refrigerate until ready to serve.
    \end{enumerate}
  \end{minipage}
}
