\renewcommand{\myUrltext}{https://www.thekitchn.com/recipe-tangy-cucumber-salad-243623}
\renewcommand{\myServingstext}{4 to 6}
\renewcommand{\mycourse}{Salads}

\hypersetup{
  pdftitle = {\myTitletext},
  pdfauthor = {\myAuthor}

\sbox{\Ingredients}{%
  \begin{minipage}[t]{\the\dimexpr 0.4\textwidth - 2em}
    \par\mySubtitle{Ingredients}
    \par\begin{itemize}
      \item \nicefrac{1}{4} cup apple cider vinegar
      \item 2 tablespoons olive oil
      \item 2 teaspoons granulated sugar
      \item 1 teaspoon kosher salt, plus more as needed
      \item Freshly ground black pepper
      \item 2 pounds cucumbers (about 4 medium)
      \item 2 tablespoons finely chopped fresh chives
    \end{itemize}
  \end{minipage}
}

\sbox{\Instructions}{%
  \begin{minipage}[t]{\the\dimexpr 0.6\textwidth - 2em}
    \par\mySubtitle{Instructions}
    \par\begin{enumerate}
      \item Place the vinegar, oil, sugar, salt, and a few grinds of pepper in a large
        bowl and whisk to combine. Thinly slice the cucumbers crosswise. Place them in the
        bowl, add the chives, and toss to combine.
      \item Cover and refrigerate for at least 30 minutes or overnight to allow the
        flavors to meld. Taste and season with salt and pepper as needed before serving.
    \end{enumerate}
  \end{minipage}
}
