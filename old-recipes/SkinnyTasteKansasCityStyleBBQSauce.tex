\renewcommand{\mytitle}{Homemade Kansas City Style BBQ Sauce (Skinny Taste)}
\renewcommand{\myurl}{https://www.skinnytaste.com/homemade-kansas-style-bbq-sauce/}

\sbox{\ingredients}{%
  \begin{minipage}[t]{0.32\textwidth}
    {\sffamily\bfseries Ingredients}\vspace{0.5ex}
    \begin{itemize}
      \item 1 \nicefrac{1}{3} cups tomato sauce
      \item \nicefrac{1}{3} cup tomato paste
      \item \nicefrac{1}{3} cup honey
      \item \nicefrac{1}{2} cup plus 3 tbsp red wine vinegar
      \item \nicefrac{1}{4} cup molasses, unsulfured
      \item 2 tsp all natural hickory liquid smoke, Colgin
      \item \nicefrac{1}{2} tsp cayenne pepper
      \item \nicefrac{1}{2} tsp salt
      \item \nicefrac{1}{2} tsp coarsely ground black pepper
      \item \nicefrac{1}{2} tsp paprika
      \item \nicefrac{1}{4} tsp garlic powder
      \item \nicefrac{1}{4} tsp onion powder
      \item \nicefrac{1}{8} tsp ground cinnamon
      \item \nicefrac{1}{8} tsp chili powder
    \end{itemize}
  \end{minipage}
}

\sbox{\method}{%
  \begin{minipage}[t]{0.55\textwidth}
    {\sffamily\bfseries Method}\vspace{0.5ex}
    \begin{enumerate}
      \item In a large saucepan combine all the ingredients and simmer
        over low heat for 25 minutes stirring occasionally.
      \item Let it cool and store in the refrigerator until ready to use.
      \item For using on grilled meat, brush onto chicken, pork or beef the
        last 10 minutes of cooking.
    \end{enumerate}
  \end{minipage}
}
