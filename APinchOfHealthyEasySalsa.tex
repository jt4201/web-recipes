\renewcommand{\mytitle}{Easy Homemade Salsa (A Pinch of Healthy)}
\renewcommand{\myurl}{https://www.apinchofhealthy.com/easy-homemade-salsa/}

\sbox{\ingredients}{%
  \begin{minipage}[t]{0.32\textwidth}
    {\sffamily\bfseries Ingredients}\vspace{0.5ex}
    \begin{itemize}
      \item 14 ounce can diced tomatoes undrained
      \item 4 ounce can of green chiles undrained
      \item \nicefrac{1}{4} cup fresh cilantro leaves (optional)
      \item 1 chunk yellow onion roughly chopped (About 2 Tablespoons )
      \item 1 clove large of garlic roughly chopped
      \item 1 tablespoon fresh lime juice (you can sub 1 Tablespoon white vinegar, if you need to)
      \item \nicefrac{1}{2} teaspoon salt or more, to taste
      \item \nicefrac{1}{4} teaspoon ground cumin
    \end{itemize}
  \end{minipage}
}

\sbox{\method}{%
  \begin{minipage}[t]{0.55\textwidth}
    {\sffamily\bfseries Method}\vspace{0.5ex}
    \begin{enumerate}
      \item Add all ingredients to the blender or food processor, and secure the lid.
      \item Press "pulse" 7 to 9 times (or more, until desired consistency), being careful not to over blend.
      \item Remove the lid, and serve.
    \end{enumerate}
  \end{minipage}
}
