\renewcommand{\mytitle}{Club Sandwich (Spend with Pennies)}
\renewcommand{\myurl}{https://www.spendwithpennies.com/club-sandwich/}

\myRecipe{Sandwiches}{\mytitle{}}

\sbox{\ingredients}{%
  \begin{minipage}[t]{\the\dimexpr 0.35\linewidth - 1em\relax}
    {\sffamily\bfseries Ingredients}\vspace{0.5ex}
    \begin{itemize}
      \item 6 slices bread toasted and buttered
      \item 4 tablespoons mayonnaise
      \item 4 ounces sliced turkey
      \item 4 ounces sliced ham
      \item 2 slices cheddar cheese
      \item 4 slices cooked bacon
      \item 1 tomato sliced
      \item \nicefrac{1}{2} cup lettuce washed and chopped
    \end{itemize}
  \end{minipage}
}

\sbox{\method}{%
  \begin{minipage}[t]{\the\dimexpr 0.6\linewidth - 1em\relax}
    {\sffamily\bfseries Method}\vspace{0.5ex}
    \begin{enumerate}
      \item Arrange 2 slices of bread on a cutting board. Spread each slice with mayonnaise.
      \item Top with turkey, tomato slices and cheddar cheese.
      \item Spread mayonnaise on 2 more pieces of bread and place on the cheddar cheese.
      \item Top with ham, bacon and lettuce. Spread mayonnaise on the final 2 slices of bread and
        place on top.
      \item Cut sandwich into halves or quarters. Serve with pickles.
    \end{enumerate}
  \end{minipage}
}
