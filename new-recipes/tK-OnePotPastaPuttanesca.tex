\sbox{\Ingredients}{%
  \begin{minipage}[t]{\the\dimexpr 0.4\textwidth - 2em}
    \par\mySubtitle{Ingredients}
    \par\begin{itemize}
      \item 12 ounces cherry tomatoes
      \item \nicefrac{1}{2} cup pitted Castelvetrano olives
      \item \nicefrac{1}{2} cup fresh parsley leaves, plus more for garnish
      \item 4 \nicefrac{1}{2} cups water
      \item 12 ounces dried linguine pasta
      \item \nicefrac{1}{4} cup capers
      \item 2 tablespoons olive oil
      \item 1 teaspoon kosher salt
      \item \nicefrac{1}{2} teaspoon freshly ground black pepper
    \end{itemize}
  \end{minipage}
}

\sbox{\Instructions}{%
  \begin{minipage}[t]{\the\dimexpr 0.6\textwidth - 2em}
    \par\mySubtitle{Instructions}
    \par\begin{enumerate}
      \item Halve 12 ounces cherry tomatoes and \nicefrac{1}{2} cup pitted Castelvetrano olives, and
        place both in a large, straight-sided skillet. Coarsely chop \nicefrac{1}{2} cup fresh parsley
        leaves and add to the skillet. Add 4 \nicefrac{1}{2} cups water, 12 ounces linguine, \nicefrac{1}{4} cup
        capers, 2 tablespoons olive oil, 1 teaspoon kosher salt, and \nicefrac{1}{2} teaspoon black
        pepper.
      \item Bring to a boil over high heat (this will take about 5 minutes). Boil the
        mixture, stirring and turning the pasta with tongs to prevent sticking, until the
        pasta is al dente and almost all the liquid has evaporated, 8 to 10 minutes.
      \item Remove from the heat. Top with more fresh parsley. Taste and season with more
        salt as needed. 
    \end{enumerate}
  \end{minipage}
}
