\sbox{\Ingredients}{%
  \begin{minipage}[t]{\the\dimexpr 0.4\textwidth - 2em}
    \par\mySubtitle{\bfseries\large Ingredients}
    \par\begin{itemize}
      \item 2 lbs Chicken Breast
      \item 1/2 cup Fresh squeezed Orange Juice
      \item 1/3 cup Fresh squeezed lime juice
      \item 3 cloves garlic
      \item 1/4 cup Cilantro (chopped)
      \item 1 lime zest of 1 lime
      \item 1/2 TBSP Cumin
      \item 1 tsp Dried Oregano
      \item 1 tsp Salt
      \item 1/2 tsp Black Pepper
      \item 1/2 cup Olive Oil
      \item 1 onion (quartered)
      \item 1 cup Quinoa (I used Harmony blend)
      \item 2 cups Water
      \item 1/4 cup cilantro, diced
      \item 1 15.5 oz can Goya Black Beans
      \item 2 tbsp Goya olive oil
      \item 1/2 onion, finely chopped
      \item 1/4 cup Bell Peppers, diced
      \item 2 cloves Garlic, minced
      \item 3/4 cup water
      \item 1 tsp Oregano
      \item 1 packet Goya Sazon (without annatto)
      \item 1 tbsp White Cooking wine or Cider Vinegar
      \item 1/2 cup Pineapple, diced
      \item 2 Tomatoes , large
      \item 1 Onions, diced
      \item 1/4 cup Cilantro (diced)
      \item 1/2 lime juiced
    \end{itemize}
  \end{minipage}
}

\sbox{\Instructions}{%
  \begin{minipage}[t]{\the\dimexpr 0.6\textwidth - 2em}
    \par\mySubtitle{\bfseries\large Instructions}
    \par\begin{enumerate}
      \item Wash and cut Chicken into bite size pieces, set aside.
      \item Juice oranges and limes. Add orange juice, lime juice, garlic, cilantro, lime
        zest, cumin, oregano, salt and black pepper.
      \item Process for 10 seconds. Add olive oil and process for additional 5 seconds
        till just combined.
      \item Pour over chicken in a container and marinade overnight or for at least 4
        hours.
      \item For Black Beans -Heat oil in medium saucepan on medium heat.
      \item Add onion, pepper, garlic; cook until tender (8-10 mins). Stir in remaining
        ingredients and bring to boil. Reduce and simmer for about 10 minutes.
      \item For Quinoa and add it along with water to a pot. Bring to a boil then reduce
        heat to medium, cover and cook for 15 minutes till water is absorbed.
      \item Fluff Quinoa and add 1/4 cup Cilantro and 1/2 juiced lime and mix in. Add salt
        to taste
      \item For Pineapple Pico de Gallo â€“ Dice Pineapple, Tomatoes 1/4 Cilantro and 1
        whole Onion.
      \item Mix to combine add lime juice and salt to taste (optional).
      \item Peel and dice Avocados. Squeeze a bit of lime juice to keep from browning and
        add salt to taste.
      \item Add a layer of Quinoa on the bottom of the bowl. Add all other toppings as you
        like. Enjoy. 
    \end{enumerate}
  \end{minipage}
}
