\sbox{\Ingredients}{%
  \begin{minipage}[t]{\the\dimexpr 0.4\textwidth - 2em}\raggedright\sffamily\small
    \par\mySubtitle{\bfseries\large Ingredients}
    \par\begin{itemize}
      \item 1 tsp Dried Minced Onion
      \item \nicefrac{1}{4} tsp Garlic Powder
      \item \nicefrac{1}{2} tsp Dried Mustard ground
      \item \nicefrac{1}{4} tsp Chili Powder (salt free)
      \item \nicefrac{1}{4} tsp black pepper
      \item 1 pound Ground Beef
      \item \nicefrac{1}{2} cup Water
      \item 1 cup Hunt's No Salt Added Ketchup (or my ketchup recipe)
    \end{itemize}
  \end{minipage}
}

\sbox{\Instructions}{%
  \begin{minipage}[t]{\the\dimexpr 0.6\textwidth - 2em}\raggedright\sffamily\small
    \par\mySubtitle{\bfseries\large Instructions}
    \par\begin{enumerate}
      \item Brown 1 pound of lean ground beef over medium heat 8 to 10 minutes in a large
        nonstick skillet or until beef is not pink, breaking beef up into small crumbles.
        Pour off drippings.
      \item Add spices, \nicefrac{1}{2} cup of water and 1 cup of no-salt-added ketchup.
      \item Simmer for 10 minutes, reducing heat as needed to maintain a gentle simmer.
        Stir occasionally. Add more water if the mixture becomes too thick. Serve hot. 
    \end{enumerate}
  \end{minipage}
}
