\sbox{\Ingredients}{%
  \begin{minipage}[t]{\the\dimexpr 0.4\textwidth - 2em}
    \par\mySubtitle{Ingredients}
    \par\begin{itemize}
      \item 4 onions, cut into \nicefrac{1}{4} inch thick half rounds
      \item \nicefrac{1}{2} cup water
      \item 2 tablespoons brown sugar
      \item 2 tablespoons balsamic vinegar
      \item 2 tablespoons olive oil
      \item pinch salt
    \end{itemize}
  \end{minipage}
}

\sbox{\Instructions}{%
  \begin{minipage}[t]{\the\dimexpr 0.6\textwidth - 2em}
    \par\mySubtitle{Instructions}
    \par\begin{enumerate}
      \item In a large saute pan with a lid, heat the oil. 
      \item Add the onions, cover and cook on low, stirring occasionally, until
        translucent but not browned. 
      \item Sprinkle with sugar, cover and continue to cook, stirring occasionally, for
        about 30 minutes or onions are golden and soft. 
      \item Add half of the water, sprinkle with salt, cover and continue to cook on low,
        stirring occasionally for another 20 minutes or until the onions are a rich, dark
        brown. 
      \item Add the remaining water and the vinegar, cook 10=15 additional minutes,
        uncovered, or until the water and vinegar has been absorbed. 
      \item Serve as is, or store in a air tight jar and refrigerate up to 1 week.
    \end{enumerate}
  \end{minipage}
}
