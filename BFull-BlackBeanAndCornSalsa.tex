\sbox{\Ingredients}{%
  \begin{minipage}[t]{\the\dimexpr 0.4\textwidth - 2em}
    \par\mySubtitle{Ingredients}
    \par\begin{description}
      \item[FOR THE SALSA] \hfill
      \begin{itemize}
        \item 15 ounce can black beans , rinsed and drained
        \item 14 ounce can sweet corn , drained (see notes for fresh or frozen corn)
        \item \nicefrac{1}{2} a small red bell pepper , ribs and seeds removed, very finely diced
        \item 1 small jalapeno , ribs and seeds removed, very finely diced
        \item \nicefrac{1}{3} cup very finely diced red onion
        \item 1 ripe avocado , diced small
    \end{itemize}
  \item[FOR THE DRESSING] \hfill
      \begin{itemize}
        \item 2 tablespoons finely chopped fresh cilantro (or parsley)
        \item 2 tablespoons extra-virgin olive oil
        \item 2 tablespoons lime juice
        \item \nicefrac{1}{2} tablespoon balsamic vinegar (or red wine vinegar)
        \item 2 teaspoons honey
        \item \nicefrac{1}{2} teaspoon coarse salt
        \item \nicefrac{1}{4} teaspoon pepper
        \item \nicefrac{1}{8} teaspoon cumin
    \end{itemize}
  \item[FOR SERVING] \hfill
      \begin{itemize}
        \item tortilla chips , optional
    \end{itemize}
\end{description}
  \end{minipage}
}

\sbox{\Instructions}{%
  \begin{minipage}[t]{\the\dimexpr 0.6\textwidth - 2em}
    \par\mySubtitle{Instructions}
    \par\begin{enumerate}
      \item Place corn, beans, bell pepper, jalapeno, and onion, in a large bowl.
      \item Prepare the dressing by adding all of the ingredients into a jar with a
        tight-fitting lid. Screw on the jar and shake vigorously for 1 minute until well
        combined and emulsified.
      \item Pour the dressing over the corn and bean mixture and gently toss to coat.
      \item The salad is okay to serve immediately, but is even better if covered and
        placed in the refrigerator for 1 hour.
      \item Just before serving, gently fold in the diced avocado.
      \item Enjoy as a side salad or transfer to a serving bowl and enjoy with tortilla
        chips.
    \end{enumerate}
  \end{minipage}
}
