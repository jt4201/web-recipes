\renewcommand{\mytitle}{Canned Salmon Salad Sandwich (Healthful Ideas)}
\renewcommand{\myurl}{https://thehealthfulideas.com/canned-salmon-salad-sandwich/}

\sbox{\ingredients}{%
  \begin{minipage}[t]{0.32\textwidth}
    {\sffamily\bfseries Ingredients}\vspace{0.5ex}
    \begin{itemize}
      \item 2 cans of salmon, drained
      \item 200g soft goat cheese or cream cheese
      \item 1-2 stalks of celery, finely chopped
      \item 1 small red onion, finely chopped
      \item \nicefrac{1}{4} cup capers, roughly chopped
      \item \nicefrac{1}{4} cup finely chopped fresh dill
      \item 2 tbsp finely chopped fresh chives
      \item 2 tbsp extra virgin olive oil
      \item 1-2 teaspoon hot sauce or chili garlic sauce
      \item 1 clove garlic, grated or minced
      \item 2 tsp dijon mustard
      \item \nicefrac{1}{2} tsp freshly ground black pepper
      \item \nicefrac{1}{4} tsp sea salt or to taste
      \item \textbf{For serving}: toasted bread, tomatoes, cucumber, lettuce, extra hot sauce, balsamic glaze
    \end{itemize}
  \end{minipage}
}

\sbox{\method}{%
  \begin{minipage}[t]{0.55\textwidth}
    {\sffamily\bfseries Method}\vspace{0.5ex}
    \begin{enumerate}
      \item Add all the ingredients into a large bowl and mix until everything is well combined.
      \item Add more herbs, hot sauce, dijon, or olive oil to taste. Season with salt and pepper.
      \item Toast your bread and spread the salmon salad on both slices. Layer with sliced tomato, cucumber, and lettuce. Drizzle with more hot sauce and balsamic glaze if desired. Enjoy!
    \end{enumerate}
  \end{minipage}
}
