\sbox{\Ingredients}{%
  \begin{minipage}[t]{\the\dimexpr 0.4\textwidth - 2em}
    \par\mySubtitle{Ingredients}
    \par\begin{itemize}
      \item 2 tablespoons unsalted butter
      \item 2 tablespoons olive oil
      \item 4 medium red or sweet onions, diced (about 4 cups)
      \item 2 tablespoons fresh thyme leaves, or 1 \nicefrac{1}{2} teaspoons dried
      \item 1 bay leaf
      \item 1 teaspoon kosher salt
      \item Freshly ground black pepper
      \item \nicefrac{1}{4} cup apple cider vinegar
      \item 2 tablespoons packed brown sugar
      \item \textbf{optional}
        \begin{itemize}
          \item 1 roasted garlic clove, mashed (optional)
        \end{itemize}
    \end{itemize}
  \end{minipage}
}

\sbox{\Instructions}{%
  \begin{minipage}[t]{\the\dimexpr 0.6\textwidth - 2em}
    \par\mySubtitle{Instructions}
    \par\begin{enumerate}
      \item Melt the butter and olive oil in a heavy-bottomed pot over medium heat. Add
        the onions, thyme, bay leaf, garlic, salt, and a few grinds of pepper. Cover and
        cook over medium-low heat until very soft, about 20 minutes.
      \item Uncover, add the vinegar and sugar, and cook, stirring frequently, until
        thickened, 20 minutes or longer. Remove the bay leaf.
      \item Transfer jam to a bowl and let cool. Taste and season with more salt and
        pepper as needed.
    \end{enumerate}
  \end{minipage}
}
