\sbox{\ingredients}{%
  \begin{minipage}[t]{\the\dimexpr 0.35\linewidth - 1em\relax}
    {\sffamily\bfseries Ingredients}\vspace{0.5ex}
    \begin{itemize}
      \item \nicefrac{1}{4} cup water
      \item 1 cup ketchup
      \item \nicefrac{1}{4} cup dried porcini mushrooms
      \item \nicefrac{1}{4} cup Worcestershire Sauce
      \item \nicefrac{1}{4} cup fresh lemon juice
      \item \nicefrac{1}{4} cup distilled white vinegar
      \item 3 tablespoons Thai or Vietnamese fish sauce
      \item 1 small white onions (or 1 large shallot)
      \item 1 clove garlic
      \item 3 tablespoons reduced sodium soy sauce
      \item 2 tablespoons firmly packed brown sugar
      \item 1 tablespoon dry mustard
    \end{itemize}
  \end{minipage}
}

\sbox{\method}{%
  \begin{minipage}[t]{\the\dimexpr 0.6\linewidth - 1em\relax}
    {\sffamily\bfseries Method}\vspace{0.5ex}
    \begin{enumerate}
      \item \textbf{Prep}. Finely chop enough onion to create \nicefrac{1}{2} cup (57.6g). Mince the garlic.
      \item \textbf{Cook}. In a saucepan, whisk together all the ingredients, bring to a boil, reduce the
        heat, partially cover, and gently simmer until the sauce is thick, chunky, and very
        aromatic, about 45 minutes.
      \item Pass it through a fine-mesh sieve to remove the solids.
      \item \textbf{Serve}. Chill then serve or refrigerate for up to 2 months.
    \end{enumerate}
  \end{minipage}
}
