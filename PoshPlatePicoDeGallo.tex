\renewcommand{\myurl}{https://poshplate.us/salsa-no-cilantro/}

\myRecipe{Sauces!Salsa}{\mytitle{}}

\sbox{\ingredients}{%
  \begin{minipage}[t]{\the\dimexpr 0.35\linewidth - 1em\relax}
    {\sffamily\bfseries Ingredients}\vspace{0.5ex}
    \begin{itemize}
      \item 3 large beefsteak tomatoes or heirloom tomatoes medium dice (about 3 cups)
      \item 1 large serrano roughly chopped (remove the seeds and stems for less heat)
      \item 1 large garlic clove minced
      \item \nicefrac{1}{2} cup red onion small dice
      \item 2 limes 2 tbsp of juice
      \item \nicefrac{1}{4} tsp cumin
      \item \nicefrac{1}{4} tsp kosher salt
    \end{itemize}
  \end{minipage}
}

\sbox{\method}{%
  \begin{minipage}[t]{\the\dimexpr 0.6\linewidth - 1em\relax}
    {\sffamily\bfseries Method}\vspace{0.5ex}
    \begin{enumerate}
      \item In a large bowl, add the roughly chopped serrano pepper, minced garlic, diced red onion,
        lime juice, cumin, and kosher salt. Mix to incorporate. While the ingredients are
        marinating, dice the tomatoes.
      \item Add the diced tomatoes over a strainer (of some sort) to help drain off excess liquid.
        When ready, add it to the rest of the ingredients. Give a toss and place it into the forge
        and allow it to chill for 30 minutes before serving.
      \item If you are absolutely too eager and cannot wait, feel free to enjoy right away.
    \end{enumerate}
  \end{minipage}
}
