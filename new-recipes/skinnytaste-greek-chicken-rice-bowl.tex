\sbox{\myIngred}{%
  \begin{minipage}[t]{\the\dimexpr0.35\textwidth}
    \textbf{Ingredients}
    \begin{jsubitem}
      \item 1 ½ pounds boneless, skinless chicken breasts (2 large or 3 small)
      \item 2 tablespoons fresh lemon juice
      \item 2 tablespoons red wine vinegar
      \item 1 tablespoon extra virgin olive oil
      \item 3 cloves garlic, minced
      \item 2 teaspoons dried oregano
      \item 1-¼ cups quick cooking brown rice, such as Uncle Bens 10 min rice
      \item 2 cups low sodium chicken broth
      \item 1 dry pint cherry or grape tomatoes
      \item 1 medium cucumber, peeled, seeded and cut into bite-sized pieces
      \item 1 small orange bell pepper, cut into 1-inch pieces
      \item 20 pitted Kalamata olives, sliced
      \item 4 teaspoons extra virgin olive oil
      \item 1 medium lemon, quartered
      \item ¼ cup crumbled feta cheese
    \end{jsubitem}
  \end{minipage}
}

\sbox{\myMethod}{%
  \begin{minipage}[t]{\the\dimexpr0.6\textwidth-2em}
    \textbf{Method}
    \begin{jitem}
    \item For the chicken
      \begin{jnum}
      \item Combine lemon juice, vinegar, olive oil, garlic and oregano in a small
        bowl.
      \item Place the chicken breasts, 1 at a time, in a Ziploc bag. Pound with a
        mallet or rolling pin to an even thickness, about ½ inch thick, being careful
        not to puncture the bag.
      \item Add marinade to the bag. Allow to marinate for at least 30 minutes or up
        to overnight in the refrigerator.
      \item Bring the rice and broth to a boil. Reduce heat to simmer, cover and cook
        10 to 12 minutes (or until rice is tender and broth is absorbed).
      \item Meanwhile preheat a grill, grill pan or heavy skillet over medium heat.
        Spray with oil and add the chicken, discard the marinade.
      \item Cook 5 to 6 minutes per side or until chicken is cooked through. Allow
        chicken to rest while you make the rest of the bowl.
      \end{jnum}
      \item To assemble bowls and eat immediately
      \begin{jnum}
      \item Slice the chicken breasts. Place ¾ cup rice in each bowl, top with
        chicken, ¼ of the tomatoes, cucumbers, pepper and olives.
      \item Drizzle with 1 teaspoon oil, lemon juice from 1 wedge, 1 tablespoon feta
        cheese and a sprinkle of parsley.
      \end{jnum}
      \item For meal prep
      \begin{jnum}
      \item Slice the chicken breasts. In 4 meal containers, layer ¾ cup rice and
        ¼ of the chicken, in separate containers or baggies pack ¼ of the tomatoes,
        cucumbers, bell pepper, olives and feta.
      \item When ready to eat, heat the chicken and rice, top with veggies, olives and
        cheese and drizzle with 1 teaspoon olive oil and lemon juice from 1 wedge.
      \end{jnum}
    \end{jitem}
  \end{minipage}
}
