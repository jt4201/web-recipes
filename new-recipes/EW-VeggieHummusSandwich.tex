\sbox{\Ingredients}{%
  \begin{minipage}[t]{\the\dimexpr 0.4\textwidth - 2em}
    \par\mySubtitle{Ingredients}
    \par\begin{itemize}
      \item 2 slices whole-grain bread
      \item 3 tablespoons hummus
      \item ¼ avocado, mashed
      \item ½ cup mixed salad greens
      \item ¼ medium red bell pepper, sliced
      \item ¼ cup sliced cucumber
      \item ¼ cup shredded carrot
    \end{itemize}
  \end{minipage}
}

\sbox{\Instructions}{%
  \begin{minipage}[t]{\the\dimexpr 0.6\textwidth - 2em}
    \par\mySubtitle{Instructions}
    \par\begin{enumerate}
      \item Spread 1 slice of bread with hummus and the other with avocado. Fill the
        sandwich with greens, bell pepper, cucumber and carrot. Slice in half and serve. 
    \end{enumerate}
  \end{minipage}
}
