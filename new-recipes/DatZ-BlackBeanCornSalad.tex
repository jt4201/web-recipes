\sbox{\Ingredients}{%
  \begin{minipage}[t]{\the\dimexpr 0.4\textwidth - 2em}
    \par\mySubtitle{Ingredients}
    \par\begin{itemize}
      \item 2 cans black beans drained and rinsed
      \item 1 \nicefrac{1}{2} cups corn kernels fresh, frozen or canned
      \item \nicefrac{1}{4} cup red onion minced
      \item 1 red bell pepper diced
      \item 1 avocado peeled, pit removed and diced
      \item 1 jalapeno ribs and seeds removed, then minced
      \item \nicefrac{1}{3} cup cilantro leaves chopped
      \item \nicefrac{1}{3} cup olive oil
      \item \nicefrac{1}{4} cup lime juice
      \item 2 teaspoons honey
      \item 1 teaspoon chili powder
      \item 1 teaspoon cumin
      \item salt and pepper to taste
    \end{itemize}
  \end{minipage}
}

\sbox{\Instructions}{%
  \begin{minipage}[t]{\the\dimexpr 0.6\textwidth - 2em}
    \par\mySubtitle{Instructions}
    \par\begin{enumerate}
      \item Place the black beans, corn, red onion, red bell pepper, avocado and jalapeno
        in a large bowl.
      \item In a small bowl, whisk together the cilantro, olive oil, lime juice, honey,
        chili powder, cumin and salt and pepper.
      \item Pour the dressing over the beans and vegetables and toss gently to coat.
        Serve.
    \end{enumerate}
  \end{minipage}
}
