\renewcommand{\mytitle}{Honey Mustard Sauce (Budget Bytes)}
\renewcommand{\myurl}{https://www.budgetbytes.com/honey-mustard-sauce/}

\sbox{\ingredients}{%
  \begin{minipage}[t]{0.32\textwidth}
    {\sffamily\bfseries Ingredients}\vspace{0.5ex}
    \begin{itemize}
      \item \nicefrac{1}{2} cup mayonnaise
      \item 1 Tbsp Dijon mustard
      \item 1 Tbsp yellow mustard
      \item 3 Tbsp honey
      \item 1 tsp apple cider vinegar
      \item \nicefrac{1}{8} tsp garlic powder
      \item \nicefrac{1}{8} tsp paprika
      \item \nicefrac{1}{8} tsp salt
      \item freshly cracked pepper (about 5 cranks of a pepper mill)
    \end{itemize}
  \end{minipage}
}

\sbox{\method}{%
  \begin{minipage}[t]{0.55\textwidth}
    {\sffamily\bfseries Method}\vspace{0.5ex}
    \begin{enumerate}
      \item Combine all the ingredients in a bowl and stir until smooth. Use immediately or
        refrigerate until ready to use.
    \end{enumerate}
  \end{minipage}
}
