\sbox{\Ingredients}{%
  \begin{minipage}[t]{\the\dimexpr 0.4\textwidth - 2em}
    \par\mySubtitle{Ingredients}
    \par\begin{itemize}
      \item 1 cup uncooked long-grain white rice
      \item 1 tablespoon extra virgin olive oil
      \item 1 small onion, chopped
      \item \nicefrac{1}{2} green bell pepper, thinly sliced
      \item \nicefrac{1}{2} red bell pepper, thinly sliced (for added flavor, you can substitute
        roasted red bell peppers or some chopped pimentos)
      \item 3 cloves garlic, minced
      \item 2 (16-ounce) cans black beans, rinsed and drained
      \item 2 tablespoons white vinegar
      \item A few dashes of Tabasco or \nicefrac{1}{4} teaspoon cayenne powder
      \item 1 teaspoon dried oregano or 1 heaping tablespoon fresh oregano, chopped
      \item Kosher salt and freshly ground black pepper, to taste
      \item Lime wedges and cilantro for garnish, optional
    \end{itemize}
  \end{minipage}
}

\sbox{\Instructions}{%
  \begin{minipage}[t]{\the\dimexpr 0.6\textwidth - 2em}
    \par\mySubtitle{Instructions}
    \par\begin{description}
    \item[Cook the rice]
    Cook rice according to package instructions. White long-grain rice usually takes 15 minutes to cook once the water is simmering, and 10 minutes to sit and steam.
    \item[Sauté the onions, bell peppers, and garlic]
    Heat oil in a large skillet on medium high. Sauté onions and bell peppers for 3 to 4 minutes, until just beginning to soften, Then, add garlic and sauté a minute more.
    \item[Add the beans, Tabasco, and vinegar, then simmer]
    Add the black beans, vinegar, and Tabasco or cayenne. Add vinegar, and bring to a simmer. Reduce heat, cover, and let simmer for 5 more minutes.
    \item[Stir in the rice and oregano]
    Stir in the rice and oregano. Salt and pepper to taste. Sprinkle with chopped cilantro and serve with lime wedges on the side, if you wish. 
    \end{description}
  \end{minipage}
}
