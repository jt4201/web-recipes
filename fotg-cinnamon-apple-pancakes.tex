\renewcommand{\myTitleText}{Cinnamon Apple Pancakes}
\renewcommand{\myURLText}{https://www.freshoffthegrid.com/cinnamon-apple-pancakes/}
\renewcommand{\myFilename}{fotg-cinnamon-apple-pancakes.png}
\renewcommand{\myServings}{8 pancakes}
\renewcommand{\myAuthor}{Fresh Off The Grid}
\renewcommand{\myCookTime}{25 minutes}
\renewcommand{\myPrepTime}{5 minutes}
\renewcommand{\myTotalTime}{30 minutes}
\renewcommand{\myPreamble}{
  Light, fluffy, and layered with incredible apple flavor, these cinnamon apple cider
  pancakes are the ultimate fall weather camping breakfast. If you like spiced apple
  cider, you’re going to LOVE these pancakes! 
}

\sbox{\myIngred}{
  \begin{minipage}[t]{\the\dimexpr 0.35\textwidth}\raggedright
    \textbf{Ingredients}
    \begin{itemize}
      \item 1 cup flour
      \item 2 tablespoons sugar
      \item 1½ teaspoon baking powder
      \item ½ teaspoon salt
      \item 1 egg
      \item ½ cup whole milk
      \item ½ cup spiced apple cider or apple juice
      \item ½ cup shredded apple
      \item Butter or oil to grease pan
      \item Maple syrup for serving
      \item \textbf{Cinnamon Apple Topping}
    \begin{itemize}
      \item 1 apple cored and cubed
      \item 1 tablespoon butter
      \item 1 teaspoon sugar
      \item 1 teaspoon cinnamon
    \end{itemize}
    \end{itemize}
  \end{minipage}
}

\sbox{\myMethod}{
  \begin{minipage}[t]{\the\dimexpr 0.55\textwidth - 2em}\raggedright
    \textbf{Method}
    \begin{enumerate}
      \item To make the batter, whisk egg, milk, and cider together in a medium bowl.
        Add the flour, sugar, baking powder, and salt, stir to combine. Fold in the
        shredded apple.
      \item Heat a skillet or griddle over medium heat. Generously grease the surface,
        then pour the batter out ¼ cup at a time.
      \item When the bottom of the pancakes are golden brown and little air bubbles
        begin to form on the top, flip the pancakes and cook until golden brown.
        Repeat with the rest of the batter.
      \item To make the Cinnamon Apple Topping, melt 1 tablespoon butter in the
        skillet and add sliced apples, sugar, and cinnamon. Stir to coat the apples,
        then saute for a few minutes until apples have softened.
      \item Serve pancakes topped with cinnamon apples and maple syrup.
    \end{enumerate}
  \end{minipage}
}
