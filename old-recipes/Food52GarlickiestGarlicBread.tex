\renewcommand{\mytitle}{Garlickiest Garlic Bread (Food52)}
\renewcommand{\myurl}{https://food52.com/recipes/77125-garlickiest-garlic-bread}

\sbox{\ingredients}{%
  \begin{minipage}[t]{0.32\textwidth}
    {\sffamily\bfseries Ingredients}\vspace{0.5ex}
    \begin{description}
      \item[Garlic Confit] \hfill
        \begin{itemize}
          \item  2 heads garlic, broken into cloves, smashed, if you please, and peeled
          \item \nicefrac{1}{2} cup unsalted butter
        \end{itemize}
      \item[Garlic Bread] \hfill
        \begin{itemize}
          \item  1 lengthy, airy loaf Italian bread (roughly 9 x 4 inches in length)
          \item 4 garlic cloves, smashed and peeled
          \item \nicefrac{1}{2} teaspoon kosher salt
          \item \nicefrac{1}{4} cup parsley leaves, roughly chopped
          \item 2 teaspoons grated pecorino or Parmesan
        \end{itemize}
    \end{description}
  \end{minipage}
}

\sbox{\method}{%
  \begin{minipage}[t]{0.55\textwidth}
    {\sffamily\bfseries Method}\vspace{0.5ex}
    \begin{description}
      \item[Garlic Confit] Melt the butter over low heat in a small saucepan
        and add the garlic. Gently simmer until the garlic and butter are both
        golden—about 45 minutes—stirring every so often to make sure no one is
        cooking too fast. Remove from the heat and let cool for a few minutes.
      \item [Garlic Bread] \hfill
        \begin{enumerate}
          \item Meanwhile, halve the bread horizontally. Score into 1 \nicefrac{1}{2}-inch wide pieces—so, slashing most of the way, but not all of the way through—for tearing apart later. Preheat the oven to 400° F. Line a rimmed sheet pan with a silicone mat or parchment or non-stick foil.
          \item Add the garlic confit (including all that butter!) to a food processor along with the raw garlic and salt. Pulse until mostly smooth. Add the parsley and pulse a couple times just to incorporate. Spread this butter mixture all over the cut-side of the bread. Add the bread to the sheet tray with the buttered side facing up.
          \item Bake for about 10 minutes until the bread is toasty and the top is colorful and brown. Immediately sprinkle with the grated cheese. Serve warm.
        \end{enumerate}
    \end{description}
  \end{minipage}
}
