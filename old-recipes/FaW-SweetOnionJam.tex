\sbox{\Ingredients}{%
  \begin{minipage}[t]{\the\dimexpr 0.4\textwidth - 2em}
    \par\mySubtitle{Ingredients}
    \par\begin{itemize}
      \item 1 tablespoon unsalted butter
      \item 2 medium Vidalia or other sweet onions, cut into \nicefrac{3}{4}-inch dice
        (about 2 \nicefrac{1}{2} cups)
      \item \nicefrac{1}{2} cup sugar
      \item \nicefrac{1}{3} cup dry red wine
      \item \nicefrac{1}{3} cup red wine vinegar
      \item \nicefrac{1}{3} cup grenadine
      \item Salt
    \end{itemize}
  \end{minipage}
}

\sbox{\Instructions}{%
  \begin{minipage}[t]{\the\dimexpr 0.6\textwidth - 2em}
    \par\mySubtitle{Instructions}
    \par\begin{enumerate}
      \item Melt the butter in a medium skillet. 
      \item Add the onions and cook over moderate heat, stirring, until soft and just
        translucent, about 8 minutes. 
      \item Add the sugar, wine, vinegar, grenadine and a pinch of salt. 
      \item Cook over low heat, stirring, until the liquid thickens and coats the back of
        a spoon, about 35 minutes. 
      \item Transfer the onion jam to a bowl and let cool.
    \end{enumerate}
  \end{minipage}
}
