\renewcommand{\mytitle}{Spicy Canned Salmon Salad Rice Bowl (Skinny Taste)}
\renewcommand{\myurl}{https://www.skinnytaste.com/spicy-canned-salmon-rice-bowl/}

\myRecipe{Lunch}{\mytitle{}}

\sbox{\ingredients}{%
  \begin{minipage}[t]{\the\dimexpr 0.35\linewidth - 1em\relax}
    {\sffamily\bfseries Ingredients}\vspace{0.5ex}
    \begin{itemize}
      \item 1 5-ounce can skinless wild pink or red salmon, in water drained
      \item 1 tablespoons light mayonnaise
      \item 2 teaspoons sriracha sauce, plus more for topping
      \item 2 scallions, white and greens separated
      \item Pinch kosher salt
      \item \nicefrac{3}{4} cup cooked brown rice, heated
      \item \nicefrac{1}{2} cup chopped cucumber, I use Persian cucumbers
      \item Furikake or chopped nori and sesame seeds, for garnish
    \end{itemize}
  \end{minipage}
}

\sbox{\method}{%
  \begin{minipage}[t]{\the\dimexpr 0.6\linewidth - 1em\relax}
    {\sffamily\bfseries Method}\vspace{0.5ex}
    \begin{enumerate}
      \item In a small bowl, combine the salmon, mayonnaise, sriracha, scallion whites and pinch of salt and mix well.
      \item In the bowl you plan to serve in, add the rice.
      \item Top it with the salmon salad, chopped cucumber, furikake and scallion greens plus more sriracha if desired.
    \end{enumerate}
  \end{minipage}
}
