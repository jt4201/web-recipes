\renewcommand{\myUrltext}{https://blog.modernistpantry.com/recipes/foodie-favorite-nacho-cheese/}
\renewcommand{\myServingstext}{1.5 cups}
\renewcommand{\mycourse}{Sauces\&Condiments!Cheese}

  pdftitle = {\myTitletext},

\sbox{\Ingredients}{%
  \begin{minipage}[t]{\the\dimexpr 0.4\textwidth - 2em}\raggedright\sffamily\small
    \par\mySubtitle{\bfseries\large Ingredients}
    \par\begin{itemize}
      \item 200g (6.8 oz) Cheddar
      \item 200g (7 fl oz.) 2\% Milk
      \item 30g (1 small) Roasted Jalapeno, Seeded \& Chopped
      \item 4.5g (1 tsp) Sodium Citrate
      \item 1.5g (\nicefrac{1}{2} tsp) Salt
    \end{itemize}
  \end{minipage}
}

\sbox{\Instructions}{%
  \begin{minipage}[t]{\the\dimexpr 0.6\textwidth - 2em}\raggedright\sffamily\small
    \par\mySubtitle{\bfseries\large Instructions}
    \begin{description}
      \item[Combine and Melt] \hfill
      \par\begin{enumerate}
        \item In a small pot over low heat, combine all ingredients listed above. Gently
          cook until cheese is melted.
    \end{enumerate}
    \item[Blend, Cool, and Serve] \hfill
    \par\begin{enumerate}
      \item Using an immersion blender, blend the contents of the pan until smooth and
        uniform.
      \item Allow the cheese sauce to cool before serving. Cheese can be cooled completely
        and reheated for future use.
  \end{enumerate}
\end{description}
\end{minipage}
}
